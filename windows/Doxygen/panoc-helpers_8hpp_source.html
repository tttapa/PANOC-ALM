21 ) {
22  // g(x)
23  p.g(x, ŷ);
24  // ζ = g(x) + Σ⁻¹y
25  ŷ += Σ.asDiagonal().inverse() * y;
26  // d = ζ - Π(ζ, D)
27  ŷ = projecting_difference(ŷ, p.D);
28  // dᵀŷ, ŷ = Σ d

33  }

190 inline real_t descent_lemma(
192  const Problem &problem,
198  real_t rounding_tolerance,
202  real_t L_max,
204  crvec xₖ,
206  real_t ψₖ,
208  crvec grad_ψₖ,
210  crvec y,
212  crvec Σ,
214  rvec x̂ₖ,
216  rvec pₖ,
218  rvec ŷx̂ₖ,
220  real_t &ψx̂ₖ,
222  real_t &norm_sq_pₖ,
224  real_t &grad_ψₖᵀpₖ,
226  real_t &Lₖ,
228  real_t &γₖ) {
229 
230  real_t old_γₖ = γₖ;
231  real_t margin = (1 + std::abs(ψₖ)) * rounding_tolerance;
232  while (ψx̂ₖ - ψₖ > grad_ψₖᵀpₖ + 0.5 * Lₖ * norm_sq_pₖ + margin) {
233  if (not(Lₖ * 2 <= L_max))
234  break;
235 
236  Lₖ *= 2;
237  γₖ /= 2;
238 
239  // Calculate x̂ₖ and pₖ (with new step size)
240  calc_x̂(problem, γₖ, xₖ, grad_ψₖ, /* in ⟹ out */ x̂ₖ, pₖ);
241  // Calculate ∇ψ(xₖ)ᵀpₖ and ‖pₖ‖²
242  grad_ψₖᵀpₖ = grad_ψₖ.dot(pₖ);
243  norm_sq_pₖ = pₖ.squaredNorm();
244 
245  // Calculate ψ(x̂ₖ) and ŷ(x̂ₖ)
246  ψx̂ₖ = calc_ψ_ŷ(problem, x̂ₖ, y, Σ, /* in ⟹ out */ ŷx̂ₖ);
247  }
248  return old_γₖ;
249 }
