317  real_t old_ψ = ψₖ;

320  bool sufficient_decrease;
321  if (0) // From paper

323  else // Since we compute ψ(x̂ₖ) we might as well pick the best one
324  sufficient_decrease = ψₖ₊₁ <= ψx̂ₖ;
325

337  grad_ψₖ.swap(grad_ψx̂ₖ);
338  }

344  // TODO: is this a valid test?
