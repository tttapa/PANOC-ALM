37  // Modified Gram-Schmidt to make q orthogonal to Q
38  q = v;
39  for (size_t i = 0; i < q_idx; ++i) {
40  real_t s = Q.col(i).dot(q);
41  r(i) = s;
42  q -= s * Q.col(i);
43  }
44 
45  // Compute the norms of orthogonalized q and original v
46  real_t norm_q = q.norm();
47  real_t norm_v = v.norm();
48 
49  // If ‖q‖ is significantly smaller than ‖v‖, perform
50  // reorthogonalization
51  real_t η = 0.7;
52  while (norm_q < η * norm_v) {
53  ++reorth_count;
54  for (size_t i = 0; i < q_idx; ++i) {
55  real_t s = Q.col(i).dot(q);
56  r(i) += s;
57  q -= s * Q.col(i);
58  }
59  norm_v = norm_q;
60  norm_q = q.norm();
61  }

64  // orthonormal columns)
65  r(q_idx) = norm_q;
66  q /= norm_q;

74  void remove_column() {
75  assert(q_idx > 0);
76 
77  // After removing the first colomn of the upper triangular matrix R,
78  // it becomes upper Hessenberg. Givens rotations are used to make it
79  // triangular again.
80  Eigen::JacobiRotation<real_t> G;
81  size_t r = 0; // row index of R
82  size_t c = r_succ(r_idx_start); // column index of R in storage
83  while (r < q_idx - 1) {
84  // Compute the Givens rotation that makes the subdiagonal element
85  // of column c or R zero.
86  G.makeGivens(R(r, c), R(r + 1, c), &R(r, c));
87  // Apply it to the remaining columns of R.
88  // Not the current column, because the diagonal element was updated
89  // by the makeGivens function, and the subdiagonal element doesn't
90  // have to be set to zero explicitly, it's implicit.
91  // Also not the previous columns, because they are already
92  // implicitly zero below the diagonal and this rotation wouldn't
93  // have any effect there.
94  // TODO: can this be sped up by applying it in two blocks instead
95  // of column by column?
96  for (size_t cc = r_succ(c); cc != r_idx_end; cc = r_succ(cc))
97  R.col(cc).applyOnTheLeft(r, r + 1, G.adjoint());
98  // Apply the inverse of the Givens rotation to Q.
99  Q.block(0, 0, Q.rows(), q_idx).applyOnTheRight(r, r + 1, G);
100  // Advance indices to next diagonal element of R.
101  ++r;
102  c = r_succ(c);
103  }
104  // Remove rightmost column of Q, since it corresponds to the bottom row
105  // of R, which was set to zero by the Givens rotations
106  --q_idx;
107  // Remove the first column of R.
108  r_idx_start = r_succ(r_idx_start);
109  }

112  template <class VecB, class VecX>
113  void solve_col(const VecB &b, VecX &x) const {
114  // Iterate over the diagonal of R, starting at the bottom right,
115  // this is standard back substitution
116  // (recall that R is stored in a circular buffer, so R.col(i) is
117  // not the mathematical i-th column)
118  auto rev_bgn = ring_reverse_iter().begin();
119  auto rev_end = ring_reverse_iter().end();
120  auto fwd_end = ring_iter().end();
121  for (auto it_d = rev_bgn; it_d != rev_end; ++it_d) {
122  // Row/column index of diagonal element of R
123  auto [rR, cR] = *it_d;
124  // (r is the zero-based mathematical index, c is the index in
125  // the circular buffer)
126  x(rR) = Q.col(rR).transpose() * b; // Compute rhs Qᵀb
127  // In the current row of R, iterate over the elements to the
128  // right of the diagonal
129  // Iterating from left to right seems to give better results
130  for (auto it_c = it_d.forwardit; it_c != fwd_end; ++it_c) {
131  auto [rX2, cR2] = *it_c;
132  x(rR) -= R(rR, cR2) * x(rX2);
133  }
134  x(rR) /= R(rR, cR); // Divide by diagonal element
135  }
136  }
