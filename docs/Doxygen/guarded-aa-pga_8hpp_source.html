245  // Calculate ∇ψ(xₖ)ᵀpₖ and ‖pₖ‖²

250  grad_ψₖᵀpₖ, Lₖ, γₖ);

253  if (γₖ != old_γₖ) {

255  // Save the latest function evaluation gₖ at the first index
256  size_t newest_g_idx = qr.ring_tail();
257  if (newest_g_idx != 0)

259  // Flush everything else and reset indices

269  }

289  if (stop_status != SolverStatus::Unknown) {

291  // It saves 1 evaluation of g per ALM iteration, but requires
292  // many extra stores in the inner loops of PANOC.
293  // TODO: move the computation of ẑ and g(x) to ALM?
294  if (stop_status == SolverStatus::Converged ||
295  stop_status == SolverStatus::Interrupted ||
296  always_overwrite_results) {

308  // Standard gradient descent step
309  gₖ = xₖ - γₖ * grad_ψₖ;
310  rₖ = gₖ - yₖ;
311 
312  // Solve Anderson acceleration least squares problem and update history
313  minimize_update_anderson(qr, G, rₖ, rₖ₋₁, gₖ, γ_LS, yₖ);
314 
315  // Project accelerated step onto feasible set
316  xₖ = project(yₖ, problem.C);
317 
318  // Calculate the objective at the projected accelerated point
319  real_t ψₖ₊₁ = calc_ψ_ŷ(xₖ, /* in ⟹ out */ ŷₖ);
320  real_t old_ψ = ψₖ;
321 
322  // Check sufficient decrease condition for Anderson iterate
323  bool sufficient_decrease;
324  if (0) // From paper
325  sufficient_decrease = ψₖ₊₁ <= ψₖ - 0.5 * γₖ * grad_ψₖ.squaredNorm();
326  else // Since we compute ψ(x̂ₖ) we might as well pick the best one
327  sufficient_decrease = ψₖ₊₁ <= ψx̂ₖ;
328 
329  if (sufficient_decrease) {
330  // Accept Anderson step
331  // yₖ and xₖ are already overwritten by yₑₓₜ and Π(yₑₓₜ)
332  ψₖ = ψₖ₊₁;
333  calc_grad_ψ_from_ŷ(xₖ, ŷₖ, /* in ⟹ out */ grad_ψₖ);
334  }
335  // If not satisfied, take normal projected gradient step
336  else {
337  yₖ.swap(gₖ);
338  xₖ.swap(x̂ₖ);
339  ψₖ = ψx̂ₖ;
340  grad_ψₖ.swap(grad_ψx̂ₖ);
341  }
342  rₖ.swap(rₖ₋₁);
343  s.accelerated_steps_accepted += sufficient_decrease;
344 
345  // Check if we made any progress, prevents us from exceeding the maximum
346  // number of iterations doing nothing if the step size gets too small
347  // TODO: is this a valid test?
348  no_progress = (ψₖ == old_ψ) ? no_progress + 1 : 0;
349  }
350  throw std::logic_error("[AAPGA] loop error");
