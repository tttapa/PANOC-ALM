181  }
182  if (not std::isfinite(Lₖ)) {

184  return s;

188  unsigned no_progress = 0;

190  // Main loop
191  // =========================================================================
192  for (unsigned k = 0; k <= params.max_iter; ++k) {
193  // From previous iteration:
194  // - xₖ
195  // - grad_ψₖ
196  // - ψₖ
197 
198  // Quadratic upper bound -----------------------------------------------
199 
200  // Projected gradient step: x̂ₖ and pₖ
201  calc_x̂(γₖ, xₖ, grad_ψₖ, /* in ⟹ out */ x̂ₖ, pₖ);
202  // Calculate ψ(x̂ₖ) and ŷ(x̂ₖ)
203  real_t ψx̂ₖ = calc_ψ_ŷ(x̂ₖ, /* in ⟹ out */ ŷₖ);
204  // Calculate ∇ψ(xₖ)ᵀpₖ and ‖pₖ‖²
205  real_t grad_ψₖᵀpₖ = grad_ψₖ.dot(pₖ);
206  real_t pₖᵀpₖ = pₖ.squaredNorm();
207 
208  // Decrease step size until quadratic upper bound is satisfied
209  descent_lemma(xₖ, ψₖ, grad_ψₖ, x̂ₖ, pₖ, ŷₖ, ψx̂ₖ, pₖᵀpₖ, grad_ψₖᵀpₖ, Lₖ,
210  γₖ);
211 
212  // Calculate ∇ψ(x̂ₖ)
213  calc_grad_ψ_from_ŷ(x̂ₖ, ŷₖ, /* in ⟹ out */ grad_ψx̂ₖ);
214 
215  // Check stop condition ------------------------------------------------
216 
217  real_t εₖ = detail::calc_error_stop_crit(params.stop_crit, pₖ, γₖ, xₖ,
218  grad_ψx̂ₖ, grad_ψₖ, problem.C);
219 
220  // Print progress
221  if (params.print_interval != 0 && k % params.print_interval == 0)
222  print_progress(k, ψₖ, grad_ψₖ, pₖ, γₖ, εₖ);
223  if (progress_cb)

225  γₖ, εₖ, Σ, y, problem, params});

228  bool out_of_time = time_elapsed > params.max_time;
229  bool out_of_iter = k == params.max_iter;
230  bool interrupted = stop_signal.stop_requested();
231  bool not_finite = not std::isfinite(εₖ);
232  bool conv = εₖ <= ε;
233  bool max_no_progress = no_progress > 1;
234  bool exit = conv || out_of_iter || out_of_time || not_finite ||
235  interrupted || max_no_progress;
236  if (exit) {
237  // TODO: We could cache g(x) and ẑ, but would that faster?
238  // It saves 1 evaluation of g per ALM iteration, but requires
239  // many extra stores in the inner loops.
240  // TODO: move the computation of ẑ and g(x) to ALM?
241  if (conv || interrupted || always_overwrite_results) {
242  calc_err_z(x̂ₖ, /* in ⟹ out */ err_z);
243  x = std::move(x̂ₖ);
244  y = std::move(ŷₖ);
245  }
246  s.iterations = k; // TODO: what do we count as an iteration?
247  s.ε = εₖ;
248  s.elapsed_time = duration_cast<microseconds>(time_elapsed);
249  s.status = conv ? SolverStatus::Converged
250  : out_of_time ? SolverStatus::MaxTime
251  : out_of_iter ? SolverStatus::MaxIter
252  : not_finite ? SolverStatus::NotFinite
253  : max_no_progress ? SolverStatus::NoProgress
254  : SolverStatus::Interrupted;
255  return s;
256  }
257 
258  if (xₖ == x̂ₖ) // TODO: this is quite expensive to do on each iteration
259  ++no_progress;
260  else
261  no_progress = 0;
262 
263  xₖ.swap(x̂ₖ);
264  grad_ψₖ.swap(grad_ψx̂ₖ);
265  ψₖ = ψx̂ₖ;
266  }
